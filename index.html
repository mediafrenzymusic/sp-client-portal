window.downloadZip = async (filter = 'all') => {

  const toDownload = filter === 'approved'

    ? currentImages.filter(i => i.status === 'approved')

    : currentImages;


  if (!toDownload.length) {

    showToast(filter === 'approved' ? 'No approved images yet' : 'No images to download');

    return;

  }


  const galleryName = currentGallery ? currentGallery.name.replace(/[^a-z0-9]/gi,'_') : 'gallery';

  const zipName = filter === 'approved' ? `${galleryName}_approved` : galleryName;


  // Show progress overlay

  const overlay = document.createElement('div');

  overlay.className = 'zip-overlay';

  overlay.innerHTML = `<div class="zip-card">

    <div class="zip-title">Preparing your download</div>

    <div class="zip-sub">Packaging ${toDownload.length} ${filter === 'approved' ? 'approved ' : ''}image${toDownload.length !== 1 ? 's' : ''} into a ZIP file…</div>

    <div class="zip-track"><div class="zip-fill" id="zip-fill" style="width:0%"></div></div>

    <div class="zip-count" id="zip-count">0 of ${toDownload.length}</div>

  </div>`;

  document.body.appendChild(overlay);


  try {

    const zip = new JSZip();

    const folder = zip.folder(zipName);

    let loaded = 0;


    await Promise.all(toDownload.map(async (img, idx) => {

      try {

        const response = await fetch(img.url);

        const blob = await response.blob();

        const ext = blob.type.includes('png') ? 'png' : blob.type.includes('webp') ? 'webp' : 'jpg';

        const filename = img.filename || `${zipName}_${String(idx+1).padStart(3,'0')}.${ext}`;

        folder.file(filename, blob);

      } catch(e) {

        folder.file(`image_${idx+1}_url.txt`, img.url);

      }

      loaded++;

      const pct = Math.round(loaded / toDownload.length * 100);

      const fill = document.getElementById('zip-fill');

      const count = document.getElementById('zip-count');

      if (fill) fill.style.width = pct + '%';

      if (count) count.textContent = `${loaded} of ${toDownload.length}`;

    }));


    const zipBlob = await zip.generateAsync({ type:'blob', compression:'DEFLATE', compressionOptions:{level:3} });

    const url = URL.createObjectURL(zipBlob);

    const a = document.createElement('a');

    a.href = url;

    a.download = `${zipName}.zip`;

    a.click();

    URL.revokeObjectURL(url);

    notifyAdmin('download_all', 0, `ZIP download (${filter})`);

    showToast(`✓ ZIP downloaded — ${toDownload.length} image${toDownload.length !== 1 ? 's' : ''}`);

  } catch(err) {

    showToast('Something went wrong — try downloading individually');

    console.error(err);

  } finally {

    document.body.removeChild(overlay);

  }

};

// ─── Shared link access (no login required) ───────────────────────────────────

async function checkSharedLink() {

  const params = new URLSearchParams(location.search);

  const token = params.get('g');

  if (!token) return false;


  // Find gallery by shareToken

  const q = query(collection(db, 'galleries'), where('shareToken', '==', token));

  const snap = await getDocs(q);

  const matched = snap.docs[0];

  if (!matched) {

    document.getElementById('loading').style.display = 'none';

    document.getElementById('auth-screen').style.display = 'flex';

    document.getElementById('auth-err').textContent = 'This share link is invalid or has expired.';

    return true; // handled

  }


  // Store gallery data for use after name prompt

  const galleryData = matched.data();

  const galleryId = matched.id;


  // Check if we already know this visitor's name (stored per share token)

  const storedName = localStorage.getItem('visitor-name-' + token);


  function openSharedGallery(visitorName) {

    document.getElementById('loading').style.display = 'none';

    document.getElementById('auth-screen').style.display = 'none';

    document.getElementById('name-prompt-modal').style.display = 'none';

    document.getElementById('app').style.display = 'block';

    document.getElementById('nav-user').textContent = visitorName;


    // Show banner

    const banner = document.getElementById('link-banner');

    document.getElementById('link-banner-name').textContent = galleryData.name;

    banner.style.display = 'block';


    // Hide admin controls and back button

    document.getElementById('admin-badge').style.display = 'none';

    document.getElementById('admin-create-wrap').style.display = 'none';

    document.getElementById('sign-out-btn').style.display = 'none';

    const backBtn = document.querySelector('.back-btn');

    if (backBtn) backBtn.style.display = 'none';


    currentGallery = { ...galleryData, id: galleryId };

    currentImages = currentGallery.images || [];

    document.getElementById('gv-title').textContent = currentGallery.name;

    document.getElementById('gv-sub').textContent = (currentGallery.shootDate || '') + ' · ' + currentImages.length + ' images';

    document.getElementById('gallery-list').style.display = 'none';

    document.getElementById('gallery-view').style.display = 'block';

    renderImages();

    updateProgress();

    subscribeToGallery(galleryId);

  }


  window.submitNamePrompt = () => {

    const name = document.getElementById('name-prompt-input').value.trim();

    if (!name) {

      document.getElementById('name-prompt-input').style.borderColor = 'var(--gold)';

      document.getElementById('name-prompt-input').focus();

      return;

    }

    localStorage.setItem('visitor-name-' + token, name);

    openSharedGallery(name);

  };


  if (storedName) {

    // Returning visitor — skip the prompt

    openSharedGallery(storedName);

  } else {

    // First visit — show the name prompt, set up the app in the background

    document.getElementById('loading').style.display = 'none';

    document.getElementById('auth-screen').style.display = 'none';

    document.getElementById('name-prompt-modal').style.display = 'flex';

    setTimeout(() => document.getElementById('name-prompt-input').focus(), 100);

  }


  return true; // handled — skip normal auth flow

}
